# 교환(swap)

```cpp
#include <iostream>

using namespace std;

void MySwap1(int a, int b)
{
    int tmp = a;
    a = b;
    b = tmp;
    cout<< "a: " << a << " b: " << b << endl;
}

void MySwap2(int& a, int& b)
{
    int tmp = a;
    a = b;
    b = tmp;
}

void MySwap3(int* a, int* b)
{
    int tmp = *a;
    *a = *b;
    *b = tmp;
}

int main()
{
    //Swap
    int a = 3;
    int b = 2;
    cout<< "a: " << a << " b: " << b << endl;
    MySwap3(&a, &b);
    cout<< "a: " << a << " b: " << b << endl;
    return 0;
}
```
